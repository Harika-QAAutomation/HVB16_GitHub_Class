#!/bin/bash

echo "Fetching all AWS regions..."
REGIONS=$(aws ec2 describe-regions --query "Regions[].RegionName" --output text)

echo
echo "Running EC2 instances across all regions:"
echo "-----------------------------------------"

FOUND=0

for REGION in $REGIONS; do
  INSTANCES=$(aws ec2 describe-instances \
    --region "$REGION" \
    --filters "Name=instance-state-name,Values=running" \
    --query "Reservations[].Instances[].InstanceId" \
    --output text)

  if [ -n "$INSTANCES" ]; then
    FOUND=1
    echo
    echo "Region: $REGION"
    for ID in $INSTANCES; do
      echo "  - $ID"
    done
  fi
done

if [ $FOUND -eq 0 ]; then
  echo
  echo "✅ No running EC2 instances found in any region."
  exit 0
fi

echo
echo "⚠️  WARNING: This will STOP ALL running EC2 instances listed above."
read -p "Type 'yes' to continue: " CONFIRM

if [[ "$CONFIRM" != "yes" ]]; then
  echo "❌ Operation cancelled."
  exit 1
fi

echo
echo "Stopping instances..."

for REGION in $REGIONS; do
  INSTANCE_IDS=$(aws ec2 describe-instances \
    --region "$REGION" \
    --filters "Name=instance-state-name,Values=running" \
    --query "Reservations[].Instances[].InstanceId" \
    --output text)

  if [ -n "$INSTANCE_IDS" ]; then
    aws ec2 stop-instances \
      --region "$REGION" \
      --instance-ids $INSTANCE_IDS
    echo "✅ Stop request sent for region $REGION"
  fi
done

echo
echo "✅ Done."
